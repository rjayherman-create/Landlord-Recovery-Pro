  const county = req.query.county;

    `SELECT * FROM county_data 
     WHERE state = '${state}' 
     AND county = '${county}'`
  );

  res.json(result.rows[0]);
});
🚀 STEP 4 — FETCH IN FRONTEND
const [taxRate, setTaxRate] = useState(0);

useEffect(() => {
  if (!selectedState || !selectedCounty) return;

  fetch(`/api/county-data?state=${selectedState}&county=${selectedCounty}`)
    .then(res => res.json())
    .then(data => {
      if (data) setTaxRate(data.tax_rate);
    });
}, [selectedState, selectedCounty]);
🚀 STEP 5 — UPDATE ESTIMATOR

Replace your function:

const calculateSavings = (marketValue: number, assessedValue: number) => {
  if (!marketValue || !assessedValue || !taxRate) return 0;

  if (assessedValue > marketValue) {
    const overAssessment = assessedValue - marketValue;
    return Math.round(overAssessment * taxRate);
  }

  return 0;
};
