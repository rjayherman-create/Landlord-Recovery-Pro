export function generateAppealArgument(data: any, comps: any[]) {
  if (!comps || comps.length === 0) {
    return "Based on available market data, the property appears to be over-assessed.";
  }

  const avgPrice =
    comps.reduce((sum, c) => sum + (c.price || 0), 0) / comps.length;

  const assessed = data.assessedValue;

  let argument = `The property located at ${data.address} appears to be over-assessed when compared to similar properties in the area. `;

  argument += `Recent comparable properties have an average market value of approximately $${Math.round(avgPrice).toLocaleString()}, `;

  argument += `while the current assessed value is $${Number(assessed).toLocaleString()}. `;

  argument += `This discrepancy suggests that the property is assessed above its fair market value. `;

  argument += `Therefore, a reduction in the assessed value is justified based on current market conditions and comparable sales data.`;

  return argument;
}
