router.get("/property-details", async (req, res) => {
  const address = req.query.address;

  try {
    const response = await fetch(
      `https://api.rentcast.io/v1/properties?address=${encodeURIComponent(address)}`,
      {
        headers: {
          "X-Api-Key": process.env.RENTCAST_API_KEY!,
        },
      }
    );

    const data = await response.json();

    if (!data || data.length === 0) {
      return res.json({});
    }

    const property = data[0];

    res.json({
      value: property.price || property.estimatedValue,
      beds: property.bedrooms,
      baths: property.bathrooms,
      sqft: property.squareFootage,
      lot: property.lotSize,
      year: property.yearBuilt,
      type: property.propertyType,
    });

  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Failed to fetch property data" });
  }
});
