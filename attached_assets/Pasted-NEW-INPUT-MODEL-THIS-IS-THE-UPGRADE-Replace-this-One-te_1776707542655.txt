🚀 AI BUILDER (THIS REPLACES TEXTAREA)
export async function buildCaseFromInputs(data: any) {
  const prompt = `
Build a small claims court statement from:

Case Type: ${data.caseType}
Agreement: ${data.agreement}
Problem: ${data.problem}
Date: ${data.date}
Amount: $${data.amount}

Make it:
- clear
- chronological
- court appropriate
`;

  const res = await openai.chat.completions.create({
    model: "gpt-4.1-mini",
    messages: [{ role: "user", content: prompt }],
  });

  return res.choices[0].message.content;
}
