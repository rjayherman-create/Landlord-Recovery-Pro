🚀 STEP 2 — CREATE PDF GENERATOR

Create file:

src/utils/generatePdf.ts
Paste this:
import PDFDocument from "pdfkit";

export function generateAppealPDF(data: any) {
  const doc = new PDFDocument();

  const buffers: any[] = [];

  doc.on("data", buffers.push.bind(buffers));

  doc.on("end", () => {});

  // HEADER
  doc.fontSize(20).text("Property Tax Appeal", { align: "center" });

  doc.moveDown();

  // OWNER INFO
  doc.fontSize(12).text(`Owner: ${data.ownerName}`);
  doc.text(`Email: ${data.email}`);
  doc.text(`Phone: ${data.phone}`);

  doc.moveDown();

  // PROPERTY
  doc.text(`Address: ${data.address}`);
  doc.text(`County: ${data.county}`);
  doc.text(`Municipality: ${data.municipality}`);

  doc.moveDown();

  // VALUES
  doc.text(`Current Assessment: $${data.currentAssessment}`);
  doc.text(`Estimated Market Value: $${data.marketValue}`);
  doc.text(`Requested Assessment: $${data.requestedAssessment}`);

  doc.moveDown();

  // REASON
  doc.text(`Basis: ${data.basis}`);

  doc.moveDown();

  doc.text("Generated by TaxAppeal DIY");

  doc.end();

  return new Promise((resolve) => {
    doc.on("end", () => {
      const pdfData = Buffer.concat(buffers);
      resolve(pdfData);
    });
  });
}
🚀 STEP 3 — CREATE DOWNLOAD ENDPOINT

In your backend routes:

import { generateAppealPDF } from "../utils/generatePdf";

router.get("/download/:id", async (req, res) => {
  const caseId = req.params.id;

  const result = await db.execute(
    `SELECT * FROM cases WHERE id = ${caseId}`
  );

  const caseData = result.rows[0];

  if (!caseData.paid) {
    return res.status(403).send("Payment required");
  }

  const pdf = await generateAppealPDF(caseData);

  res.setHeader("Content-Type", "application/pdf");
  res.setHeader(
    "Content-Disposition",
    "attachment; filename=appeal.pdf"
  );

  res.send(pdf);
});
