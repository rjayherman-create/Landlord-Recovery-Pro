🚀 STEP 1 — BACKEND: GET COMPS

Using your existing RentCast setup:

Add new route:

router.get("/comps", async (req, res) => {
  const address = req.query.address;

  try {
    const response = await fetch(
      `https://api.rentcast.io/v1/avm/comparables?address=${encodeURIComponent(address)}`,
      {
        headers: {
          "X-Api-Key": process.env.RENTCAST_API_KEY!,
        },
      }
    );

    const data = await response.json();

    res.json(data.comparables || []);

  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Failed to fetch comps" });
  }
});
