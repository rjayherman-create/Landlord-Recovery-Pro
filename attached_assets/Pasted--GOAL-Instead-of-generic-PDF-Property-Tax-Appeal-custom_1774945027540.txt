import { PDFDocument } from "pdf-lib";
import fs from "fs";

export async function fillRP524(data: any) {
  const existingPdfBytes = fs.readFileSync("public/forms/rp524.pdf");

  const pdfDoc = await PDFDocument.load(existingPdfBytes);

  const page = pdfDoc.getPages()[0];

  const { width, height } = page.getSize();

  const draw = (text: string, x: number, y: number) => {
    page.drawText(text || "", {
      x,
      y,
      size: 10,
    });
  };

  // 🔥 MAP YOUR FORM DATA → PDF POSITIONS

  draw(data.ownerName, 100, 650);
  draw(data.address, 100, 630);
  draw(data.county, 100, 610);

  draw(String(data.currentAssessment), 100, 550);
  draw(String(data.marketValue), 100, 530);
  draw(String(data.requestedAssessment), 100, 510);

  const pdfBytes = await pdfDoc.save();

  return pdfBytes;
}
🚀 STEP 4 — CONNECT TO DOWNLOAD ROUTE

Update your download route:

import { fillRP524 } from "../utils/fillRp524";

router.get("/download/:id", async (req, res) => {
  const caseId = req.params.id;

  const result = await db.execute(
    `SELECT * FROM cases WHERE id = ${caseId}`
  );

  const caseData = result.rows[0];

  if (!caseData.paid) {
    return res.status(403).send("Payment required");
  }

  const pdf = await fillRP524(caseData);

  res.setHeader("Content-Type", "application/pdf");
  res.setHeader("Content-Disposition", "attachment; filename=rp524.pdf");

  res.send(pdf);
});
