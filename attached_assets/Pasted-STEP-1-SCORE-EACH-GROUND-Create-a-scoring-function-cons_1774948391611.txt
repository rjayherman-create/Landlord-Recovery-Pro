STEP 1 — SCORE EACH GROUND

Create a scoring function:

const scoreGrounds = (data: any, comps: any[]) => {
  const grounds = [];

  const market = Number(data.marketValue);
  const assessed = Number(data.assessedValue);
  const diff = assessed - market;
  const percentDiff = diff / market;

  // 🔥 OVERVALUATION
  if (percentDiff > 0.1) {
    grounds.push({
      name: "Overvaluation",
      score: percentDiff * 100,
    });
  }

  // 🔥 UNEQUAL ASSESSMENT
  if (comps.length >= 2) {
    const avgComp =
      comps.reduce((sum, c) => sum + c.price, 0) / comps.length;

    if (assessed > avgComp) {
      grounds.push({
        name: "Unequal Assessment",
        score: (assessed - avgComp) / avgComp * 100,
      });
    }
  }

  return grounds.sort((a, b) => b.score - a.score);
};
