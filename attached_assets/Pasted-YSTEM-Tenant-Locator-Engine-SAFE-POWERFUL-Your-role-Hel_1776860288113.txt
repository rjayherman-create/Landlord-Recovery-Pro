🔷 FRONTEND COMPONENT (DROP-IN)
/components/TenantLocator.tsx
import { useState } from "react";

export default function TenantLocator() {
  const [leads, setLeads] = useState<any[]>([]);

  const addLead = () => {
    const value = prompt("Enter new lead:");
    if (!value) return;

    setLeads([...leads, { value, status: "unverified" }]);
  };

  return (
    <div className="p-6 bg-white rounded-xl shadow">
      <h2 className="text-xl font-bold mb-4">
        Tenant Locator
      </h2>

      <button
        onClick={addLead}
        className="bg-blue-600 text-white px-4 py-2 rounded mb-4"
      >
        Add Lead
      </button>

      <ul>
        {leads.map((lead, i) => (
          <li key={i} className="border p-2 mb-2">
            {lead.value} ({lead.status})
          </li>
        ))}
      </ul>
    </div>
  );
}
