🚀 STEP 3 — BACKEND ROUTE

Create:

src/routes/property.ts
Paste:
import { Router } from "express";
import fetch from "node-fetch";

const router = Router();

router.get("/property-value", async (req, res) => {
  const address = req.query.address;

  try {
    const response = await fetch(
      `https://api.rentcast.io/v1/properties?address=${encodeURIComponent(address)}`,
      {
        headers: {
          "X-Api-Key": process.env.RENTCAST_API_KEY!,
        },
      }
    );

    const data = await response.json();

    if (!data || data.length === 0) {
      return res.json({ value: null });
    }

    res.json({
      value: data[0].price || data[0].estimatedValue,
    });

  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Failed to fetch property data" });
  }
});

export default router;
🔌 REGISTER ROUTE

In app.ts:

useEffect(() => {
  if (address.length < 10) return;

    fetch(`/api/property-value?address=${encodeURIComponent(address)}`)
      .then(res => res.json())
      .then(data => {
        if (data.value) {
          setMarketValue(data.value);
        }
      });
  }, 800);

  return () => clearTimeout(timeout);
